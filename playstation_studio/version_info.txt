# PyInstaller Windows version resource for PlayStation Studio.
#
# An unsigned executable with NO version metadata is the worst case for
# antivirus / SmartScreen heuristics. Embedding real version info (company,
# product, description, version) makes the binary look like legitimate
# software and clears many false-positive flags. Referenced from the spec via
# EXE(version=...). Windows-only; ignored on macOS/Linux.

    ffi=FixedFileInfo(
        filevers=(1, 0, 3, 0),
        prodvers=(1, 0, 3, 0),
        mask=0x3F,
        flags=0x0,
        OS=0x40004,
        fileType=0x1,
        subtype=0x0,
        date=(0, 0),
    ),
    kids=[
        StringFileInfo(
            [
                StringTable(
                    "040904B0",
                    [

                        ),
                        StringStruct("OriginalFilename", "PlayStation Studio.exe"),
                        StringStruct("ProductName", "PlayStation Studio"),
                        StringStruct("ProductVersion", "1.0.3.0"),
                    ],
                )
            ]
        ),
        VarFileInfo([VarStruct("Translation", [0x0409, 1200])]),
    ],
)
